#include <bits/stdc++.h> 
using namespace std; 
int main() {
	int test;
	cin>>test;
	while(test--){
	    int f,s;
	    string s1,s2;
	 
	   cin>>s1;
	 cin>>s2;
 	 f=s1.length();
	 s=s2.length();
 int t[f+1][s+1];
t[0][0]=0;
for(int i=1;i<f+1;i++) t[i][0]=0;
for(int i=1;i<s+1;i++) t[0][i]=0;

	for(int i=1;i<f+1;i++) {
	    for(int j=1;j<s+1;j++){
	        if(s1[i-1]==s2[j-1]){
                t[i][j]=1+t[i-1][j-1];
                                      }
            else{
                t[i][j]=max(t[i-1][j],t[i][j-1]);
                 }
     
                    }
	  	            }   
	    cout<<(f+s-t[f][s])<<endl;
	    
	}       

	return 0;
}

2
abcd xycd

6 ---abxycd
