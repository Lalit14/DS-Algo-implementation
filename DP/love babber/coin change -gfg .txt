
class Solution
{
  public:
    long long int count( int S[], int m, int sum)
    {
    long long int t[m+1][sum+1];
        t[0][0]=0;
        for(int i=1;i<sum+1;i++){
            t[0][i]=0;
        }
       for(int i=1;i<m+1;i++){
            t[i][0]=1;
        }
for(int i=1;i<m+1;i++){
    for(int j=1;j<sum+1;j++){
        if(j>=S[i-1]){
            t[i][j]=(t[i-1][j]+t[i][j-S[i-1]]);
            
        }
        else{
            t[i][j]=t[i-1][j];
        }
    }
    
    
}
 return t[m][sum];
    }
};
